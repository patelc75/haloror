  def self.select_oldest_file_for_processing(file_names)
    if file_names.size == 1
      return file_names[0]
    end
    file_names.sort! do |afile, bfile|
      atime = get_time_in_seconds(afile)
      btime = get_time_in_seconds(bfile)
      if atime <= btime
        return afile
      else
        return bfile
      end
    end
    return file_names[0]
  end
